// Simuleert 1000 EnergyBoxes
// Berekening: Φ = Σ (cos Δφ) / N
// Detecteert veldinstabiliteit (> 0.02 fase-variance)
// Geeft waarschuwing en automatisch herstel via feedback-factor

class CoherenceTestSuite {
  constructor(nodeCount = 1000) {
    this.nodes = Array(nodeCount).fill().map(() => ({
      phase: Math.random() * 2 * Math.PI,
      coherence: 0.9 + Math.random() * 0.1,
      energy_level: 0.8 + Math.random() * 0.2
    }));
    this.fieldCoherence = 0;
    this.unstableThreshold = 0.02;
  }
  
  calculateFieldCoherence() {
    // Φ = Σ (cos Δφ) / N
    let sum = 0;
    const n = this.nodes.length;
    
    for(let i = 0; i < n; i++) {
      for(let j = i+1; j < n; j++) {
        const phaseDiff = this.nodes[i].phase - this.nodes[j].phase;
        sum += Math.cos(phaseDiff);
      }
    }
    
    this.fieldCoherence = (2 * sum) / (n * (n - 1));
    return this.fieldCoherence;
  }
  
  detectInstability() {
    const phaseVariances = [];
    const n = this.nodes.length;
    
    for(let i = 0; i < n; i++) {
      let variance = 0;
      for(let j = 0; j < n; j++) {
        if(i !== j) {
          const phaseDiff = this.nodes[i].phase - this.nodes[j].phase;
          variance += Math.pow(Math.sin(phaseDiff), 2);
        }
      }
      variance /= (n - 1);
      phaseVariances.push(variance);
    }
    
    const maxVariance = Math.max(...phaseVariances);
    return maxVariance > this.unstableThreshold;
  }
  
  applyFeedbackCorrection() {
    const refPhase = this.calculateAveragePhase();
    const feedbackStrength = 0.05;
    
    this.nodes.forEach(node => {
      const phaseDiff = refPhase - node.phase;
      node.phase += phaseDiff * feedbackStrength;
    });
    
    return this.calculateFieldCoherence();
  }
  
  calculateAveragePhase() {
    let sinSum = 0, cosSum = 0;
    
    this.nodes.forEach(node => {
      sinSum += Math.sin(node.phase);
      cosSum += Math.cos(node.phase);
    });
    
    return Math.atan2(sinSum, cosSum);
  }
  
  runTest(iterations = 100) {
    const results = {
      initialCoherence: this.calculateFieldCoherence(),
      coherenceHistory: [],
      stabilityEvents: [],
      finalCoherence: 0
    };
    
    for(let i = 0; i < iterations; i++) {
      // Simuleer natuurlijke evolutie
      this.nodes.forEach(node => {
        node.phase += (Math.random() - 0.5) * 0.1;
      });
      
      const currentCoherence = this.calculateFieldCoherence();
      results.coherenceHistory.push(currentCoherence);
      
      if(this.detectInstability()) {
        results.stabilityEvents.push({
          iteration: i,
          preCorrection: currentCoherence
        });
        
        const correctedCoherence = this.applyFeedbackCorrection();
        results.stabilityEvents[results.stabilityEvents.length - 1].postCorrection = correctedCoherence;
      }
    }
    
    results.finalCoherence = this.calculateFieldCoherence();
    return results;
  }
}

# hexOSEI Failsafe Protocol v1.0

## Detectie van Incoherentie

Een veld wordt als incoherent beschouwd wanneer:
- Φ < 0.97 (globale veldcoherentie)
- >5% van de knopen een fase-variance heeft van >0.05
- Snelle coherentie-oscillatie (>3 schommelingen per cyclus)

## Automatische Terugschakeling

1. **Fase Isolatie**
   - Identificeer de laatste stabiele fase (Φ > 0.985 gedurende >10 cycli)
   - Kopieer de parameterwaarden van deze fase naar een quarantaine-buffer

2. **Gefaseerde Terugschakeling**
   - Verminder veld-energie met 30%
   - Fasesyncs beëindigen
   - Pas golden-ratio reductie toe (φ⁻¹) op alle fase-parameters

3. **Stabilisatie Procedure**
   - Voer `stabilize()` functie uit met feedbackfactor 0.15
   - Wacht 5 volledige fase-cycli
   - Verhoog energie incrementeel met 10% per cyclus

## Ma'at Recovery Ledger

Elke failsafe-activering registreert:

```json
{
  "timestamp": "ISO8601",
  "pre_failure_coherence": 0.962,
  "failure_type": "phase_variance",
  "affected_nodes": ["E0012", "E0087", "E0145"],
  "rollback_phase": 6,
  "recovery_time_cycles": 12,
  "post_recovery_coherence": 0.988
}
```

## Herstelvalidatie

Een veld wordt als hersteld beschouwd wanneer:
1. Φ > 0.985 gedurende >3 cycli
2. Fase-variance < 0.01 voor alle knopen
3. Harmonische stabiliteit gehandhaafd bij energieverhoging
