// core/resonance-recovery-protocol.js
export class ResonanceRecoveryProtocol {
  constructor(threshold = 0.8) {
    this.threshold = threshold;
    this.recoveryCount = 0;
    this.lastAmplitude = 0.01;
  }

  recover(kappa, integrity, noise = 0) {
    if (!Array.isArray(kappa) || kappa.length === 0) return kappa;

    if (integrity < this.threshold) {
      this.recoveryCount++; 
      const amp = this.computeAmplitude(integrity, noise);
      const recovered = this.addNoise(this.normalize(kappa), amp);
      
      // Gentle feedback towards mean
      const mean = recovered.reduce((a,b) => a + b, 0) / recovered.length;
      const balanced = recovered.map(v => v * 0.7 + mean * 0.3);
      
      return balanced;
    }

    return kappa; // No recovery needed  
  }
}
  
  
// core/global-field-supervisor.js  
export class GlobalFieldSupervisor {
  constructor(multiField) {
    this.multiField = multiField;
  }
  
  analyze() {
    const kappa = this.multiField.kappa || [];
    
    const coherence = this.computeCoherence(kappa);
    const entropy = this.computeEntropy(kappa); 
    const stability = this.computeStability(kappa, coherence, entropy);
    const noise = this.computeNoiseLevel(kappa);
    const integrity = this.computeIntegrity(coherence, stability, entropy, noise);
    
    return { kappa, coherence, entropy, stability, noise, integrity };
  }
  
  computeIntegrity(coherence, stability, entropy, noise) {
    const raw = (coherence + stability + (1 - entropy)) / 3;
    const sigmoid = 1 / (1 + Math.exp(-8 * (raw - 0.5))); 
    const noiseFactor = 1 - 0.3 * noise;
    return Math.max(0, Math.min(1, sigmoid * noiseFactor));
  }
}
